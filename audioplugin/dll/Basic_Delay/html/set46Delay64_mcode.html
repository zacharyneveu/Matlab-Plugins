1classdef Basic_Delay < audioPlugin % inherit from audioPlugin base class
 2    properties      % Stores public data for class
 3        Gain = 1.5  % gain of echo
 4        Delay = 0.5 % Time in s of echo
 5    end
 6    properties (Access = private) % Store circular buffer privately
 7        CircularBuffer = zeros(192001,2);
 8        BufferIndex = 1;
 9        NSamples = 0;
10    end
11    properties (Constant) % add interface to plugin
12        PluginInterface = audioPluginInterface(...
13            audioPluginParameter('Gain',...
14            'DisplayName', 'Echo Gain',...
15            'Mapping',{'lin',0,3}),...
16            audioPluginParameter('Delay',...
17            'DisplayName', 'Echo Delay',...
18            'Label','seconds'))
19    end
20    methods
21        function out = process(plugin, in)
22            out = zeros(size(in));
23            writeIndex = plugin.BufferIndex;
24            readIndex = writeIndex - plugin.NSamples;
25            if readIndex <= 0
26                readIndex = readIndex + 192001;
27            end
28
29            for i = 1:size(in,1)
30                % write input to circular buffer at start
31                plugin.CircularBuffer(writeIndex,:) = in(i,:);
32
33                % set echo to read part of circular buffer
34                echo = plugin.CircularBuffer(readIndex,:);
35
36                % assign out=in+echo
37                out(i,:) = in(i,:) + echo * plugin.Gain;
38
39                writeIndex = writeIndex + 1;
40                % stop overflow if need be
41                if writeIndex > 192001
42                    writeIndex = 1;
43                end
44
45                readIndex = readIndex + 1;
46                % stop overflow if need be
47                if readIndex > 192001
48                    readIndex = 1;
49                end
50            end
51            plugin.BufferIndex = writeIndex;
52        end

53        function set.Delay(plugin, val)
54            plugin.Delay = val;
55            % Samp/sec * sec = samples
56            plugin.NSamples = floor(getSampleRate(plugin)*val);
57        end

58        function reset(plugin)
59            plugin.CircularBuffer = zeros(192001,2);
60            plugin.NSamples = floor(getSampleRate(plugin)*plugin.Delay);
61        end
62    end
63end
